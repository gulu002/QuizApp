# 嘉年华刷题 App 数据库建表语句

> 数据库：MySQL 8.0+  
> 字符集：utf8mb4  
> 引擎：InnoDB  

---

```sql
-- ============================================
-- 嘉年华刷题 App 数据库建表脚本
-- ============================================

-- 1. 学科分类表
CREATE TABLE `category` (
    `id`            BIGINT          NOT NULL AUTO_INCREMENT  COMMENT '分类ID',
    `name`          VARCHAR(64)     NOT NULL                 COMMENT '分类名称（如：内科护理学、外科护理学）',
    `parent_id`     BIGINT          DEFAULT 0                COMMENT '父分类ID，0为顶级',
    `sort_order`    INT             DEFAULT 0                COMMENT '排序序号',
    `create_time`   DATETIME        NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `update_time`   DATETIME        NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    PRIMARY KEY (`id`),
    INDEX `idx_parent` (`parent_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='学科分类表';


-- 2. 题目表（核心）
CREATE TABLE `question` (
    `id`                BIGINT          NOT NULL AUTO_INCREMENT  COMMENT '题目ID',
    `category_id`       BIGINT          DEFAULT NULL             COMMENT '学科分类ID，关联category表',
    `question_type`     VARCHAR(16)     NOT NULL                 COMMENT '题型：SINGLE-单选 / MULTI-多选 / JUDGE-判断',
    `title`             TEXT            NOT NULL                 COMMENT '题干',
    `options_json`      JSON            NOT NULL                 COMMENT '选项JSON，格式：[{"label":"A","text":"..."},...]',
    `correct_answer`    VARCHAR(32)     NOT NULL DEFAULT ''      COMMENT '正确答案：单选"A"/判断"A"，多选"BDE"（字母升序拼接）',
    `explanation`       TEXT            DEFAULT NULL             COMMENT '题目解析',
    `difficulty`        TINYINT         DEFAULT 1                COMMENT '难度：1-简单 2-中等 3-困难',
    `source_number`     INT             DEFAULT NULL             COMMENT '原文档题号，用于溯源',
    `status`            TINYINT         NOT NULL DEFAULT 1       COMMENT '状态：1-启用 0-禁用',
    `create_time`       DATETIME        NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `update_time`       DATETIME        NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    PRIMARY KEY (`id`),
    INDEX `idx_type` (`question_type`),
    INDEX `idx_category` (`category_id`),
    INDEX `idx_difficulty` (`difficulty`),
    INDEX `idx_source` (`source_number`),
    FULLTEXT INDEX `ft_title` (`title`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='题目表';


-- 3. 用户表（表名使用users避免H2/MySQL保留字冲突）
CREATE TABLE `users` (
    `id`                BIGINT          NOT NULL AUTO_INCREMENT  COMMENT '用户ID',
    `phone`             VARCHAR(16)     DEFAULT NULL             COMMENT '手机号',
    `password_hash`     VARCHAR(128)    DEFAULT NULL             COMMENT '密码哈希（BCrypt）',
    `wechat_openid`     VARCHAR(64)     DEFAULT NULL             COMMENT '微信OpenID',
    `wechat_unionid`    VARCHAR(64)     DEFAULT NULL             COMMENT '微信UnionID',
    `nickname`          VARCHAR(64)     DEFAULT '刷题人'         COMMENT '昵称',
    `avatar_url`        VARCHAR(512)    DEFAULT NULL             COMMENT '头像URL',
    `status`            TINYINT         NOT NULL DEFAULT 1       COMMENT '状态：1-正常 0-禁用',
    `last_login_time`   DATETIME        DEFAULT NULL             COMMENT '最后登录时间',
    `create_time`       DATETIME        NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `update_time`       DATETIME        NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    PRIMARY KEY (`id`),
    UNIQUE KEY `uk_phone` (`phone`),
    UNIQUE KEY `uk_wechat_openid` (`wechat_openid`),
    UNIQUE KEY `uk_wechat_unionid` (`wechat_unionid`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户表';


-- 4. 答题记录表
CREATE TABLE `answer_record` (
    `id`                BIGINT          NOT NULL AUTO_INCREMENT  COMMENT '记录ID',
    `user_id`           BIGINT          NOT NULL                 COMMENT '用户ID',
    `question_id`       BIGINT          NOT NULL                 COMMENT '题目ID',
    `session_id`        VARCHAR(64)     NOT NULL                 COMMENT '练习会话ID（一次练习一个会话）',
    `user_answer`       VARCHAR(32)     NOT NULL DEFAULT ''      COMMENT '用户答案：单选"A"/多选"BDE"/判断"A"或"B"',
    `is_correct`        TINYINT         NOT NULL DEFAULT 0       COMMENT '是否正确：1-正确 0-错误',
    `duration_ms`       INT             DEFAULT NULL             COMMENT '该题作答耗时（毫秒）',
    `create_time`       DATETIME        NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (`id`),
    INDEX `idx_user_session` (`user_id`, `session_id`),
    INDEX `idx_user_question` (`user_id`, `question_id`),
    INDEX `idx_user_correct` (`user_id`, `is_correct`),
    INDEX `idx_create_time` (`create_time`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='答题记录表';


-- 5. 错题本表
CREATE TABLE `wrong_question` (
    `id`                BIGINT          NOT NULL AUTO_INCREMENT  COMMENT '记录ID',
    `user_id`           BIGINT          NOT NULL                 COMMENT '用户ID',
    `question_id`       BIGINT          NOT NULL                 COMMENT '题目ID',
    `wrong_count`       INT             NOT NULL DEFAULT 1       COMMENT '累计错误次数',
    `consecutive_right` INT             NOT NULL DEFAULT 0       COMMENT '最近连续答对次数（达到阈值自动移出）',
    `first_wrong_time`  DATETIME        NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '首次错误时间',
    `last_wrong_time`   DATETIME        NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '最近错误时间',
    `last_answer_time`  DATETIME        DEFAULT NULL             COMMENT '最近一次作答时间',
    `status`            TINYINT         NOT NULL DEFAULT 1       COMMENT '状态：1-在错题本中 0-已移出',
    `create_time`       DATETIME        NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `update_time`       DATETIME        NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    PRIMARY KEY (`id`),
    UNIQUE KEY `uk_user_question` (`user_id`, `question_id`),
    INDEX `idx_user_status` (`user_id`, `status`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='错题本表';


-- 6. 收藏表
CREATE TABLE `favorite` (
    `id`                BIGINT          NOT NULL AUTO_INCREMENT  COMMENT '记录ID',
    `user_id`           BIGINT          NOT NULL                 COMMENT '用户ID',
    `question_id`       BIGINT          NOT NULL                 COMMENT '题目ID',
    `create_time`       DATETIME        NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (`id`),
    UNIQUE KEY `uk_user_question` (`user_id`, `question_id`),
    INDEX `idx_user_time` (`user_id`, `create_time`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='收藏表';


-- 7. 练习会话表
CREATE TABLE `practice_session` (
    `session_id`        VARCHAR(64)     NOT NULL                 COMMENT '会话ID（UUID）',
    `user_id`           BIGINT          NOT NULL                 COMMENT '用户ID',
    `session_type`      VARCHAR(16)     NOT NULL                 COMMENT '类型：ORDER-顺序 / RANDOM-随机 / EXAM-模拟考 / WRONG-错题 / FAV-收藏 / TYPE-题型专练',
    `question_type`     VARCHAR(16)     DEFAULT NULL             COMMENT '限定题型（题型专练时使用）',
    `category_id`       BIGINT          DEFAULT NULL             COMMENT '限定分类（分类专练时使用）',
    `total_count`       INT             NOT NULL                 COMMENT '总题数',
    `correct_count`     INT             NOT NULL DEFAULT 0       COMMENT '答对数',
    `start_time`        DATETIME        NOT NULL                 COMMENT '开始时间',
    `end_time`          DATETIME        DEFAULT NULL             COMMENT '结束时间',
    `duration_sec`      INT             DEFAULT NULL             COMMENT '总耗时（秒）',
    `status`            TINYINT         NOT NULL DEFAULT 1       COMMENT '状态：1-进行中 2-已完成',
    PRIMARY KEY (`session_id`),
    INDEX `idx_user_status` (`user_id`, `status`),
    INDEX `idx_user_start` (`user_id`, `start_time`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='练习会话表';


-- 8. 用户统计表（汇总数据，定时/异步更新）
CREATE TABLE `user_stats` (
    `id`                    BIGINT      NOT NULL AUTO_INCREMENT,
    `user_id`               BIGINT      NOT NULL                 COMMENT '用户ID',
    `total_answers`         INT         NOT NULL DEFAULT 0       COMMENT '累计答题数',
    `total_correct`         INT         NOT NULL DEFAULT 0       COMMENT '累计答对数',
    `total_wrong`           INT         NOT NULL DEFAULT 0       COMMENT '累计答错数',
    `wrong_book_count`      INT         NOT NULL DEFAULT 0       COMMENT '当前错题本数量',
    `favorite_count`        INT         NOT NULL DEFAULT 0       COMMENT '当前收藏数量',
    `total_duration_sec`    BIGINT      NOT NULL DEFAULT 0       COMMENT '累计学习时长（秒）',
    `streak_days`           INT         NOT NULL DEFAULT 0       COMMENT '连续刷题天数',
    `last_practice_date`    DATE        DEFAULT NULL             COMMENT '最近刷题日期',
    `update_time`           DATETIME    NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    PRIMARY KEY (`id`),
    UNIQUE KEY `uk_user` (`user_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户统计表';


-- 9. 每日打卡表
CREATE TABLE `daily_checkin` (
    `id`                BIGINT          NOT NULL AUTO_INCREMENT,
    `user_id`           BIGINT          NOT NULL,
    `checkin_date`      DATE            NOT NULL                 COMMENT '打卡日期',
    `answer_count`      INT             NOT NULL DEFAULT 0       COMMENT '当日答题数',
    `create_time`       DATETIME        NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (`id`),
    UNIQUE KEY `uk_user_date` (`user_id`, `checkin_date`),
    INDEX `idx_user` (`user_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='每日打卡表';


-- 10. 题目掌握度表（基于答题历史推算）
CREATE TABLE `question_mastery` (
    `id`                BIGINT          NOT NULL AUTO_INCREMENT,
    `user_id`           BIGINT          NOT NULL,
    `question_id`       BIGINT          NOT NULL,
    `total_attempts`    INT             NOT NULL DEFAULT 0       COMMENT '总作答次数',
    `correct_attempts`  INT             NOT NULL DEFAULT 0       COMMENT '答对次数',
    `consecutive_right` INT             NOT NULL DEFAULT 0       COMMENT '最近连续答对次数',
    `mastered`          TINYINT         NOT NULL DEFAULT 0       COMMENT '是否已掌握：1-是（连续答对≥3次）0-否',
    `last_answer_time`  DATETIME        DEFAULT NULL,
    `update_time`       DATETIME        NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    PRIMARY KEY (`id`),
    UNIQUE KEY `uk_user_question` (`user_id`, `question_id`),
    INDEX `idx_user_mastered` (`user_id`, `mastered`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='题目掌握度表';


-- 11. 答案批量导入临时表
CREATE TABLE `answer_import` (
    `id`                BIGINT          NOT NULL AUTO_INCREMENT,
    `source_number`     INT             NOT NULL                 COMMENT '原文档题号',
    `question_type`     VARCHAR(16)     NOT NULL                 COMMENT '题型',
    `correct_answer`    VARCHAR(32)     NOT NULL                 COMMENT '正确答案',
    `explanation`       TEXT            DEFAULT NULL             COMMENT '解析',
    `imported`          TINYINT         NOT NULL DEFAULT 0       COMMENT '是否已同步到question表：1-已同步 0-未同步',
    `create_time`       DATETIME        NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (`id`),
    INDEX `idx_imported` (`imported`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='答案批量导入临时表';
```

## 表关系总览

```
category (学科分类)
    │
    │ 1:N
    ▼
question (题目) ───────────────┐
    │                          │
    │ N:N (via answer_record)  │ N:N (via favorite)
    ▼                          ▼
  user (用户) ──────────── favorite (收藏)
    │
    ├── answer_record (答题记录) ─── practice_session (练习会话)
    ├── wrong_question (错题本)
    ├── question_mastery (掌握度)
    ├── user_stats (统计汇总)
    └── daily_checkin (每日打卡)
```

## 关键设计说明

| 设计点 | 说明 |
|---|---|
| **选项存为 JSON** | `options_json` 字段直接存选项数组，避免额外的选项子表，3110 题场景下一次查询即可获取完整题目 |
| **session 会话机制** | 每次练习生成唯一 `session_id`，所有答题记录绑定会话，便于回溯整场模拟考/练习 |
| **错题自动移出** | `consecutive_right` 累计达 3 次后 `status=0`，避免错题本无限膨胀 |
| **掌握度独立** | 与错题本解耦，二者关注不同维度 |
| **全文索引** | `question.title` 建 FULLTEXT 索引，支持题干模糊搜索 |
| **统计异步更新** | `user_stats` 由定时任务或 MQ 异步聚合，避免实时计算拖慢答题接口 |
| **答案导入流程** | Excel → `answer_import` 临时表 → 存储过程同步到 `question.correct_answer` → 清空临时表 |

## 索引策略

| 表 | 索引 | 用途 |
|---|---|---|
| `question` | `idx_type` | 按题型筛选 |
| `question` | `idx_category` | 按学科分类筛选 |
| `question` | `ft_title` | 题干关键词搜索 |
| `answer_record` | `idx_user_session` | 查询某次练习的全部记录 |
| `answer_record` | `idx_user_question` | 查询某用户对某题的答题历史 |
| `wrong_question` | `uk_user_question` | 唯一约束，防止重复插入 |
| `wrong_question` | `idx_user_status` | 查询当前错题列表 |
| `favorite` | `uk_user_question` | 唯一约束，防重复收藏 |
| `practice_session` | `idx_user_status` | 查询进行中/已完成的练习 |
| `daily_checkin` | `uk_user_date` | 唯一约束，每天只能打卡一次 |
| `question_mastery` | `uk_user_question` | 唯一约束，每人每道题一条掌握记录 |
